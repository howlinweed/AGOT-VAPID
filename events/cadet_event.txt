character_event = { 
   id = cadet.1
   desc = "EVTDESCcadet.1"
   picture = GFX_evt_courtiers_talking
   
   is_triggered_only = yes
	
	option = { #House Blackstark
		name = "EVTOPTAcadet.1"
		trigger = { not = { has_global_flag = cadet_blackstark } }
	
		set_global_flag = cadet_blackstark
		FROM = { dynasty = 450200 }
	}
	option = { #House Goldstark
		name = "EVTOPTBcadet.1"
		trigger = { not = { has_global_flag = cadet_goldstark } }

		name = "EVTOPTDcadet.1"

		set_global_flag = cadet_dreadstark
		FROM = { dynasty = 450203 }
	}
	after = {
		if = { # Fired from house customizer - father/sibling will be FROMFROMFROM
			limit = { FROM  = { character = ROOT } }
			hidden_tooltip = {
				FROMFROMFROM = { 
					add_trait = assign_mission_target
					set_dynasty_flag = has_cadet_house		
					ROOT = {
						add_trait = assign_mission_target
						set_dynasty_flag = is_cadet_house
						set_dynasty_flag = cadet_house_@PREV		
						PREV = { set_dynasty_flag = owned_cadet_house_@ROOT }
						set_dynasty_flag = has_dynamic_coa #THESE HOUSES USE DYNAMIC COA SYSTEM
						character_event = { id = dynamic_coa.1 }
					}
					reverse_opinion = { who = ROOT modifier = opinion_cadet_affection years = 200 }
					opinion = { who = ROOT modifier = opinion_cadet_affection years = 200 }

						limit = { NOT = { character = PREV } }
						reverse_opinion = { who = ROOT modifier = opinion_cadet_affection years = 200 }
						opinion = { who = ROOT modifier = opinion_cadet_affection years = 200 }

				father_even_if_dead = { has_dynasty_flag = owned_cadet_house_@PREV }

			}

		NOT = { trait = kingsguard }
		NOT = { trait = archmaester }

	}

			limit = { mother_even_if_dead = { dynasty = ROOT } }

		}

			limit = { father_even_if_dead = { dynasty = ROOT } }
			father_even_if_dead = { save_event_target_as = cadet_founder }
		}
		if = {
			limit = { 
				NOR = {
					father_even_if_dead = { dynasty = ROOT }
					mother_even_if_dead = { dynasty = ROOT }
				}
			}
			mother_even_if_dead = { save_event_target_as = cadet_founder }
			father_even_if_dead = { save_event_target_as = cadet_founder }
		}

			limit = { event_target:cadet_founder = { dynasty = ROOT } }

					dynasty = ROOT

						father_even_if_dead = { dynasty = PREV NOT = { character = ROOT } }

					}
				}
				set_character_flag = change_dynasty

					limit = { 
						dynasty = ROOT 
						NOR = {
							father_even_if_dead = { dynasty = PREV NOT = { character = ROOT } NOT = { character = PREV } }
							mother_even_if_dead = { dynasty = PREV NOT = { character = ROOT } NOT = { character = PREV } }

					}

					any_child_even_if_dead = {
						limit = { 
							dynasty = ROOT 
							NOR = {
								father_even_if_dead = { dynasty = PREV NOT = { character = ROOT } NOT = { character = PREV } }
								mother_even_if_dead = { dynasty = PREV NOT = { character = ROOT } NOT = { character = PREV } }
							}
						}
						set_character_flag = change_dynasty
					}
				}
			}
			#create character to get interesting dynasty name
			if = { 
				limit = { dynasty = event_target:cadet_founder }

					culture = THIS

					age = 30

					ROOT = { dynasty = PREV }

				}

				limit = { has_character_flag = change_dynasty }
				dynasty = ROOT
				clr_character_flag = change_dynasty

					limit = { has_character_flag = change_dynasty }
					dynasty = ROOT

					any_child_even_if_dead = {
						limit = { has_character_flag = change_dynasty }

						clr_character_flag = change_dynasty
					}
				}
			}
		}

		add_trait = assign_mission_target
		set_dynasty_flag = is_cadet_house
